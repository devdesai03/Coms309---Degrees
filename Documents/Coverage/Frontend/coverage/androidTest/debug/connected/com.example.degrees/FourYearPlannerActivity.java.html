package com.example.degrees;

import android.os.AsyncTask;
import android.os.Bundle;
import android.view.View;
import android.view.ViewGroup;
import android.widget.Button;
import android.widget.TableLayout;
import android.widget.TableRow;
import android.widget.TextView;
import androidx.appcompat.app.AppCompatActivity;
import androidx.cardview.widget.CardView;
import androidx.core.content.ContextCompat;
import org.json.JSONArray;
import org.json.JSONException;

import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.net.HttpURLConnection;
import java.net.URL;
import java.util.ArrayList;
import java.util.List;

/**
 * Manages the display of a four-year planner, fetching and displaying courses for each semester.
 */
public class FourYearPlannerActivity extends AppCompatActivity {

    // OUR URL FOR TEH API
    private static final String API_URL = "http://coms-309-036.class.las.iastate.edu:8080/";

    private Button addSemesterButton;
    private ViewGroup semesterContainer;
    private int semesterCounter = 0;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.four_year_planner_activity);

        addSemesterButton = findViewById(R.id.addSemesterButton);
        semesterContainer = findViewById(R.id.semesterContainer);

        addSemesterButton.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                addSemester("Semester " + (++semesterCounter));
            }
        });
    }

    /**
     * Adds a new semester to the display, fetching and displaying its respective courses.
     *
     * @param semesterName The name of the semester to be added.
     */
    private void addSemester(String semesterName) {
        CardView semesterCard = new CardView(this);
        semesterCard.setLayoutParams(new CardView.LayoutParams(
                ViewGroup.LayoutParams.MATCH_PARENT, ViewGroup.LayoutParams.WRAP_CONTENT));
        semesterCard.setContentPadding(16, 16, 16, 16);

        TextView semesterTitle = new TextView(this);
        semesterTitle.setText(semesterName);
        semesterTitle.setTextSize(20);
        semesterTitle.setTextColor(ContextCompat.getColor(this, R.color.BlueViolet));

        // Create a TableLayout for course rows
        TableLayout courseTable = new TableLayout(this);
        courseTable.setLayoutParams(new TableLayout.LayoutParams(
                ViewGroup.LayoutParams.MATCH_PARENT, ViewGroup.LayoutParams.WRAP_CONTENT));
        courseTable.setStretchAllColumns(true);

        // Add components to the semester container
        semesterCard.addView(semesterTitle);
        semesterCard.addView(courseTable);

        semesterContainer.addView(semesterCard);

        // Fetch and add courses from the backend
        new GetCoursesDataTask(courseTable).execute(API_URL + "/semester/" + semesterName);
    }

    // AsyncTask to get courses and add them to the semester
    private class GetCoursesDataTask extends AsyncTask<String, Void, String> {
        private TableLayout courseTable;

        public GetCoursesDataTask(TableLayout courseTable) {
            this.courseTable = courseTable;
        }

        @Override
        protected String doInBackground(String... params) {
            String apiUrl = params[0];
            try {
                URL url = new URL(apiUrl);
                HttpURLConnection connection = (HttpURLConnection) url.openConnection();

                // Set up the GET request
                connection.setRequestMethod("GET");
                connection.setDoInput(true);

                // Read the response
                BufferedReader reader = new BufferedReader(new InputStreamReader(connection.getInputStream()));
                StringBuilder response = new StringBuilder();
                String line;
                while ((line = reader.readLine()) != null) {
                    response.append(line);
                }

                reader.close();
                connection.disconnect();

                return response.toString();
            } catch (Exception e) {
                e.printStackTrace();
                return null;
            }
        }

        @Override
        protected void onPostExecute(String result) {
            if (result != null) {
                // Handle the response data and add courses to the table
                try {
                    JSONArray coursesArray = new JSONArray(result);
                    for (int i = 0; i < coursesArray.length(); i++) {
                        String courseName = coursesArray.getString(i);
                        addCourse(courseTable, courseName);
                    }
                } catch (JSONException e) {
                    e.printStackTrace();
                }
            }
        }
    }

    /**
     * Adds a course to the specified TableLayout.
     *
     * @param courseTable The TableLayout to which the course will be added.
     * @param courseName  The name of the course to be added.
     */
    private void addCourse(TableLayout courseTable, String courseName) {
        TableRow courseRow = new TableRow(this);
        courseRow.setLayoutParams(new TableRow.LayoutParams(
                ViewGroup.LayoutParams.MATCH_PARENT, ViewGroup.LayoutParams.WRAP_CONTENT));

        TextView courseNameTextView = new TextView(this);
        courseNameTextView.setText(courseName);
        courseNameTextView.setLayoutParams(new TableRow.LayoutParams(0, ViewGroup.LayoutParams.WRAP_CONTENT, 2));

        // Add components to the course row
        courseRow.addView(courseNameTextView);

        // Add the course row to the table
        courseTable.addView(courseRow);
    }
}
